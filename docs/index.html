Quick Start

// Install
npm install @aspect/e2ee-webrtc

// Use
import { isE2EESupported, KeyManager } from '@aspect/e2ee-webrtc';

// Check support
if (!isE2EESupported()) {
  console.warn('E2EE not supported');
}

// Create key manager
const keyManager = new KeyManager({
  keyHistorySize: 5,
  autoRotate: true,
  rotationIntervalMs: 30000,
});

// Generate encryption key
await keyManager.generateKey();

// Listen for events
keyManager.on('key-rotated', (data) => {
  console.log(`Key rotated to generation ${data.generation}`);
});
